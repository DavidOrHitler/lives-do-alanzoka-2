    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const mainVideoPlayer = document.getElementById('main-video-player');
            const liveTitle = document.getElementById('live-title');
            const streamerDetails = document.querySelector('.streamer-details');
            const followBtn = document.getElementById('follow-btn');
            const vodList = document.getElementById('vod-list');

            // Estado inicial do botão
            let isFollowing = false;
            followBtn.textContent = '⭐ Seguir';

            // Ação do Botão Seguir
            followBtn.addEventListener('click', () => {
                isFollowing = !isFollowing; // Inverte o estado
                if (isFollowing) {
                    followBtn.textContent = '✔️ Seguindo';
                    followBtn.style.backgroundColor = '#772ce8';
                    alert('Você agora está seguindo o Alanzoka! É o puro suco do caos.');
                } else {
                    followBtn.textContent = '⭐ Seguir';
                    followBtn.style.backgroundColor = '#9147ff';
                    alert('Você parou de seguir. Volte para o caos quando quiser!');
                }
            });

            // Função para carregar um vídeo no player principal
            function loadVideo(title, streamer, views, url) {
                // Atualiza o Player
                mainVideoPlayer.src = url;
                mainVideoPlayer.load(); // Recarrega o player com a nova fonte
                
                // Atualiza as Informações
                liveTitle.textContent = title;
                
                const viewsSpan = document.getElementById('views');
                if (viewsSpan) {
                    viewsSpan.textContent = views;
                } else {
                    // Recria a estrutura caso o elemento não exista
                    streamerDetails.innerHTML = `<span class="streamer-name">${streamer}</span> <span class="separator">|</span> <span id="views" class="views-count">${views}</span>`;
                }

                // Simula o scroll para o topo para ver o novo vídeo
                window.scrollTo({ top: 0, behavior: 'smooth' });
            }

            // Adiciona evento de clique a todos os cards de VOD
            vodList.querySelectorAll('.vod-card').forEach(card => {
                card.addEventListener('click', () => {
                    // Pega os dados do atributo 'data-'
                    const title = card.getAttribute('data-title');
                    const streamer = card.getAttribute('data-streamer');
                    const views = card.getAttribute('data-views');
                    const url = card.getAttribute('data-url');
                    
                    // Chama a função para carregar o vídeo
                    loadVideo(title, streamer, views, url);
                });
            });
            
            // Carrega o primeiro VOD ao iniciar (apenas para dar conteúdo ao player)
            const firstCard = vodList.querySelector('.vod-card');
            if(firstCard) {
                loadVideo(
                    firstCard.getAttribute('data-title'),
                    firstCard.getAttribute('data-streamer'),
                    firstCard.getAttribute('data-views'),
                    firstCard.getAttribute('data-url')
                );
            }
        });
    </script>
